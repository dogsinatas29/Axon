## 1. Atomic Commit (File System Safe)

``` python
import os
import shutil
import tempfile

def _safe_path(base_dir, filename):
    normalized = os.path.normpath(filename)
    if normalized.startswith("..") or os.path.isabs(normalized):
        raise ValueError(f"Invalid filename: {filename}")
    return os.path.join(base_dir, normalized)

def _write_file(root, filename, content):
    full_path = _safe_path(root, filename)
    os.makedirs(os.path.dirname(full_path), exist_ok=True)
    with open(full_path, "w", encoding="utf-8") as f:
        f.write(content)

def _validate_files(root, expected_files):
    for filename, _ in expected_files:
        full_path = os.path.join(root, filename)
        if not os.path.exists(full_path):
            raise RuntimeError(f"Missing file: {filename}")

def atomic_commit(files, target_dir):
    tmp_dir = tempfile.mkdtemp(prefix="axon_commit_")
    try:
        for filename, code in files:
            _write_file(tmp_dir, filename, code)

        _validate_files(tmp_dir, files)

        backup_dir = None
        if os.path.exists(target_dir):
            backup_dir = target_dir + "_backup"
            if os.path.exists(backup_dir):
                shutil.rmtree(backup_dir)
            shutil.move(target_dir, backup_dir)

        shutil.move(tmp_dir, target_dir)

        if backup_dir and os.path.exists(backup_dir):
            shutil.rmtree(backup_dir)

        return True

    except Exception:
        if os.path.exists(tmp_dir):
            shutil.rmtree(tmp_dir)

        if 'backup_dir' in locals() and backup_dir and os.path.exists(backup_dir):
            if os.path.exists(target_dir):
                shutil.rmtree(target_dir)
            shutil.move(backup_dir, target_dir)

        return False
```

## 2. Parser (3-Tier)

``` python
import re

FILE_PATTERN = re.compile(r"FILE\s*:\s*(.+)", re.IGNORECASE)

def extract_code_blocks(text):
    pattern = re.compile(r"```(?:[a-zA-Z]*)\n(.*?)```", re.DOTALL)
    return pattern.findall(text)

def parse_strict(text):
    if "[OUTPUT]" not in text:
        return None

    files = []
    blocks = text.split("FILE:")

    for block in blocks[1:]:
        lines = block.strip().splitlines()
        filename = lines[0].strip()

        code_blocks = extract_code_blocks(block)
        if not code_blocks:
            return None

        code = code_blocks[0]

        if not filename or not code.strip():
            return None

        files.append((filename, code))

    return files if files else None

def parse_relaxed(text):
    files = []
    matches = FILE_PATTERN.split(text)

    if len(matches) < 2:
        return None

    for i in range(1, len(matches), 2):
        filename = matches[i].strip()
        body = matches[i + 1]

        code_blocks = extract_code_blocks(body)

        if code_blocks:
            code = code_blocks[0]
        else:
            code = body.strip()

        if not filename or not code:
            return None

        files.append((filename, code))

    return files if files else None

def parse_heuristic(text):
    code_blocks = extract_code_blocks(text)

    if not code_blocks:
        return None

    largest = max(code_blocks, key=len)
    return [("main.py", largest)]

def parse_output(text):
    for parser in (parse_strict, parse_relaxed, parse_heuristic):
        result = parser(text)
        if result:
            return result

    return None
```
