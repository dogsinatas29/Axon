``` python
from database import get_user, save_user
from calculator import calculate_age
from input_year import input_year
from input_month import input_month
from input_day import input_day

def main():
    name = input("Enter your name: ").strip()

    record = get_user(name)

    if record:
        print(f"[INFO] Existing record found for {name}")
        year = record["birth_year"]
        month = record["birth_month"]
        day = record["birth_day"]
    else:
        year = input_year()
        month = input_month()
        day = input_day(year, month)

    result = calculate_age(year, month, day)

    save_user({
        "name": name,
        "birth_year": year,
        "birth_month": month,
        "birth_day": day
    })

    print(f"\nResult for {name}")
    print(f"Korean Age: {result['korean_age']}")
    print(f"International Age: {result['international_age']}")

if __name__ == "__main__":
    main()
```

``` python
import sqlite3

DB_NAME = "users.db"

def _connect():
    return sqlite3.connect(DB_NAME)

def _init():
    conn = _connect()
    cur = conn.cursor()
    cur.execute("""
    CREATE TABLE IF NOT EXISTS user_records (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name TEXT UNIQUE,
        birth_year INTEGER,
        birth_month INTEGER,
        birth_day INTEGER,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    )
    """)
    conn.commit()
    conn.close()

_init()

def get_user(name: str):
    conn = _connect()
    cur = conn.cursor()
    cur.execute("SELECT name, birth_year, birth_month, birth_day FROM user_records WHERE name = ?", (name,))
    row = cur.fetchone()
    conn.close()

    if row:
        return {
            "name": row[0],
            "birth_year": row[1],
            "birth_month": row[2],
            "birth_day": row[3]
        }
    return None

def save_user(data: dict):
    conn = _connect()
    cur = conn.cursor()
    cur.execute("""
    INSERT INTO user_records (name, birth_year, birth_month, birth_day)
    VALUES (?, ?, ?, ?)
    ON CONFLICT(name) DO UPDATE SET
        birth_year=excluded.birth_year,
        birth_month=excluded.birth_month,
        birth_day=excluded.birth_day
    """, (
        data["name"],
        data["birth_year"],
        data["birth_month"],
        data["birth_day"]
    ))
    conn.commit()
    conn.close()
```

``` python
from datetime import date

def calculate_age(year: int, month: int, day: int):
    today = date.today()
    birth = date(year, month, day)

    international_age = today.year - birth.year
    if (today.month, today.day) < (birth.month, birth.day):
        international_age -= 1

    korean_age = today.year - birth.year + 1

    return {
        "international_age": international_age,
        "korean_age": korean_age
    }
```

``` python
from datetime import date

def validate_year(year: int):
    current_year = date.today().year
    if year < 1900 or year > current_year:
        raise ValueError("Invalid year")
    return year

def validate_month(month: int):
    if month < 1 or month > 12:
        raise ValueError("Invalid month")
    return month

def validate_day(year: int, month: int, day: int):
    try:
        date(year, month, day)
    except ValueError:
        raise ValueError("Invalid day")
    return day
```
