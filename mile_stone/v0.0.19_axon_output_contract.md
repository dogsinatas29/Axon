from validation_cluster import ValidationCluster
from calculation_cluster import CalculationCluster
from db_management_cluster import DBManagementCluster

class Hub:
    def __init__(self):
        self.validation = ValidationCluster()
        self.calc = CalculationCluster()
        self.db = DBManagementCluster()

    def run(self, data):
        validated = self.validation.validate_input(data)
        age = self.calc.calculate_age(validated["birthdate"])
        self.db.insert_data(age, validated["name"])
        return {"age": age, "name": validated["name"]}

if __name__ == "__main__":
    hub = Hub()
    result = hub.run({
        "name": "John",
        "birthdate": "2000-01-01"
    })
    print(result)

class ValidationCluster:
    def validate_input(self, data):
        if not isinstance(data, dict):
            raise ValueError("Invalid input format.")

        if "name" not in data:
            raise ValueError("Missing 'name' field.")

        if "birthdate" not in data:
            raise ValueError("Missing 'birthdate' field.")

        return data
```

from datetime import date

class CalculationCluster:
    def calculate_age(self, birthdate):
        birthdate = date.fromisoformat(birthdate)
        today = date.today()

        age = today.year - birthdate.year

        if (today.month, today.day) < (birthdate.month, birthdate.day):
            age -= 1

        return age
```
